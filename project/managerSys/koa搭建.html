# 项目构建

  • 安装脚手架 npm install -g koa-generator
  • 生成项目 koa2 -e project(项目名称) （-e 代表使用模板引擎ejs）
  • 进入项目 安装依赖
cd project
npm i
  • 启动项目
npm run start
----
nodemon ./bin/www //nodemon 更改代码后热更新的插件

# mongodb 安装

  • 安装mongodb
  • 安装mongoose npm i mongoose --s
  • 配置mongoose

在根目录创建 config 文件夹 mkdir config 新建config脚本 cd config touch config.js

module.exports = {
    dbs:'mongodb://127.0.0.1:27017/keyubin'
}

# 路由抽离及表结构

cd root 
mkdir dbs
cd dbs
mkdir models  //表结构
mkdir controllers //控制器（逻辑代码）
mkdir router      // 路由

# 创建 user 表 Schema设计

cd dbs/models
touch Users.js

// Users.js
const mongoose = require('mongoose');
let Users = new mongoose.Schema({
    username:{
        type:String,
        required:true,
        unique:true //唯一性
    },
    password:{
        type:String,
        required:true,
    }
})
module.exports = mongoose.model('Users',Users)

# 路由配置

cd routers
touch index.js

index.js

const router = require('koa-router')()
const UserController = require('../dbs/controllers/UsersController.js');
router.post('/login',UserController.login);
module.exports = router

# app.js 中引入router.js

const index = require('./routes/index.js')
const mongoose = require('mongoose');
const dbConfig = require('./dbs/config/index.js');
app.use(index.routes(), index.allowedMethods());
mongoose.connect(dbConfig.dbs,{
  useNewUrlParser:true
})

# 跨域处理

app.use(async (ctx, next) => {
  ctx.set('Access-Control-Allow-Origin', '*');
  await next();
 });
