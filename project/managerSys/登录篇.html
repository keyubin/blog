# 查找用户

有_id 则通过_id 查找没有则 通过username 查找


const FindUser = async(req)=>{
    let res
    if(req._id){
        res = await UserSchema.findById(req._id)
    }else {
        res = await UserSchema.findOne({username:req.username})
    }
    return Promise.resolve(res);
}

# 登陆及返回token

token

登陆时验证完账号及密码，符合则返回token，使用 jsonwebtoken进行签名 ，存入该用户得信息 在进行验证得时候拿到 解密然后拿到用户信息和数据库做对比。

满足以下条件则验证通过

1.过期时间 > 当前时间
2.签名中的信息在数据库中符合
jwt.sign({_id:user._id},'secret',expiresIn:60*60*4); token信息 密匙 过期时间

const UserSchema = require('../models/Users');

const bcrypt = require('bcrypt'); //密码加密

const jwt = require('jsonwebtoken');

const Login = async(ctx)=>{
    ler req = ctx.request.body;
    let user = await FindUser(req); //先查找是否存在用户
    if(user){
        const confirmRes = bcrypt.compareSync(req.password,user.password) //密码解密 对比
        if(confirmRes){
            const token = jwt.sign({_id:user._id},'secret',expiresIn:60*60*4);
            ctx.body={code:200,msg:'登陆成功',data:{token:`Bearer ${token}`}}
        }else{
            ctx.body={code:405,msg:"密码错误"}
        }
    }else{
        ctx.body={code:405,msg:"用户名不存在"}
    }
}

# 添加token验证中间件

cd /middlewares
touch tokenCheck.js

const jwt = require('jsonwebtoken');
const util = require('util');
const verify = util.promisify(jwt.verify);
const UsersControllers = require('/dbs/controllers/UsersControllers.js');
async function check (ctx,next) {
 let url = ctx.request.url;
 if(url =='/user/login') await next(); //登陆接口不进行token验证
 else{
     let token = ctx.request.headers["authorization"];
     let payload
     try{
         payload = await verify(token.split(' ')[1],'secret') //拿到token 去掉前面的Bearear 传入密匙解密
         let hasUser = await UsersControllers.FindUser({_id:payload._id}) //在数据库找解密后得数据
         let data = Math.round(new Date().getTime()/1000);
         if(payload.exp - data>0 && hasUser){
             await next(); //验证通过
         }else{  
            ctx.body = {
                code: 50014,
                message:'token 已过期'
            };
         }
     }catch(err){
         ctx.body = {
            code: 50014,
            message:'token 错误'
          };
     }
 }
}

module.exports = check;

# 在app.js中使用验证token的中间件


const check = require('/middlewears/check');
app.use(check)
